구글 시트 연동 방법
1. sheet.new 에서 새 스프레드시트 생성
2. 주소창에서 시트 ID 복사 (…/d/여기/edit)
3. 확장 프로그램 → Apps Script 클릭
4. 아래 코드를 붙여넣고 YOUR_SHEET_ID 교체 후 저장(💾)
5. 배포 → 새 배포 → 웹앱
6. 실행: 나 / 액세스: 모든 사용자 → 배포
7. 생성된 URL을 위 칸에 붙여넣기
function doPost(e) { var ss = SpreadsheetApp.openById('YOUR_SHEET_ID'); var sheet = ss.getSheets()[0]; // 헤더 자동 생성 if (sheet.getLastRow() === 0) { sheet.appendRow([ '기록 시각','종목명','현재가','전일시가', '전일대비%','상승%','상승목표가','하락%','하락목표가' ]); sheet.getRange(1,1,1,9) .setFontWeight('bold') .setBackground('#1a73e8') .setFontColor('#ffffff'); } var d = JSON.parse(e.postData.contents); sheet.appendRow([ d.time, d.name, d.cur, d.prev, d.prevChg, d.upPct, d.upTarget, d.downPct, d.downTarget ]); return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); }
